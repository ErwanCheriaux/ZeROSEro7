202 /* Freq = channel * FREQ_STEP */
203 #define SX_CHANNEL_TO_FREQ( channel, freq ) \
204  do \
205  { \
206  uint32_t initialChanInt, initialChanFrac; \
207  initialChanInt = channel >> 8; \
208  initialChanFrac = channel - ( initialChanInt << 8 ); \
209  freq = initialChanInt * FREQ_STEP_8 + ( ( initialChanFrac * FREQ_STEP_8 + ( 128 ) ) >> 8 ); \
210  }while( 0 )
211 
212 /* channel = Freq / FREQ_STEP */
213 #define SX_FREQ_TO_CHANNEL( channel, freq ) \
214  do \
215  { \
216  uint32_t initialFreqInt, initialFreqFrac; \
217  initialFreqInt = freq / FREQ_STEP_8; \
218  initialFreqFrac = freq - ( initialFreqInt * FREQ_STEP_8 ); \
219  channel = ( initialFreqInt << 8 ) + ( ( ( initialFreqFrac << 8 ) + ( FREQ_STEP_8 / 2 ) ) / FREQ_STEP_8 ); \
220  }while( 0 )
